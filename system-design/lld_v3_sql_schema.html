leave_requests — updated status column
CREATE TABLE leave_requests (
    id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    applicant_id     UUID REFERENCES users(id),
    leave_type_id    INT  REFERENCES leave_types(id),
    start_date       DATE NOT NULL,
    end_date         DATE NOT NULL,
    total_days       INT  NOT NULL,
    reason           TEXT NOT NULL,

    -- 4 possible statuses now
    status           VARCHAR(12) DEFAULT 'PENDING'
                     CHECK (status IN ('PENDING','PROCESSING','APPROVED','REJECTED')),

    decided_by       VARCHAR(20),   -- RULE_ENGINE / AI / ADMIN
    decision_reason  TEXT,
    created_at       TIMESTAMP DEFAULT NOW(),
    decided_at       TIMESTAMP,
    CHECK (end_date >= start_date)
);

CREATE INDEX idx_leaves_user   ON leave_requests(applicant_id);
CREATE INDEX idx_leaves_status ON leave_requests(status);
